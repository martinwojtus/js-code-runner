new Vue({
  el: '#app',
  data: { count: 0 },
  methods: {
    inc() {
      this.count++
    },
    dec() {
      this.count--
    }
  },
  render() {
    return (
      <div id="app">
        <h2 class="counter">{this.count}</h2>
        <button onClick={this.inc}>inc</button>
        <button onClick={this.dec}>dec</button>
      </div>
    )
  }
})



<div id="app"></div>



button {
  color: white;
  border: 1px solid #e2e2e2;
  background: magenta;
  padding: 10px 0;
  font-size: 2rem;
  width: 200px;
}

.counter {
  font-size: 2rem;
  margin-top: 20px;
}
